deleteGlobalSecondaryIndex

deleteGlobalSecondaryIndex is a in the DynamoDB extension that deletes a global secondary index from a table. For more information, see Using Global Secondary Indexes in DynamoDB.

Uses

Run deleteGlobalSecondaryIndex

Available s

For more information, see DynamoDB API: DeleteTable and DynamoDB API: DeleteGlobalSecondaryIndexAction.

Name Type Description Requirement
tableName String

that is indexed

 Required
indexName String Required

<databaseChangeLog
    xmlns="http://www.liquibase.org/xml/ns/dbchangelog"
    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
    xmlns:dynamodb-pro="http://www.liquibase.org/xml/ns/pro-dynamodb"
    xsi:schemaLocation="http://www.liquibase.org/xml/ns/dbchangelog
    http://www.liquibase.org/xml/ns/dbchangelog/dbchangelog-latest.xsd
    http://www.liquibase.org/xml/ns/pro-dynamodb
    http://www.liquibase.org/xml/ns/pro-dynamodb/liquibase-pro-dynamodb-latest.xsd">

    <changeSet id="1" author="your.name">
        <dynamodb-pro:createDynamoTable tableName="deleteGlobalSecondaryIndexXml">
            <dynamodb-pro:keySchema hashAttributeName="Artist" rangeAttributeName="SongTitle"/>
            <dynamodb-pro:attributeDefinitions>
                <dynamodb-pro:attributeDefinition attributeName="Artist" attributeType="S"/>
                <dynamodb-pro:attributeDefinition attributeName="SongTitle" attributeType="S"/>
            </dynamodb-pro:attributeDefinitions>
            <dynamodb-pro:provisionedThroughput readCapacityUnits="10" writeCapacityUnits="10"/>
            <dynamodb-pro:globalSecondaryIndexes>
                <dynamodb-pro:globalSecondaryIndex indexName="globalIndex">
                    <dynamodb-pro:keySchema hashAttributeName="Artist" rangeAttributeName="SongTitle"/>
                    <dynamodb-pro:projection projectionType="INCLUDE">
                        <dynamodb-pro:nonKeyAttributes attributeName="Artist"/>
                        <dynamodb-pro:nonKeyAttributes attributeName="SongTitle"/>
                    </dynamodb-pro:projection>
                    <dynamodb-pro:provisionedThroughput readCapacityUnits="5" writeCapacityUnits="5"/>
                </dynamodb-pro:globalSecondaryIndex>
            </dynamodb-pro:globalSecondaryIndexes>
        </dynamodb-pro:createDynamoTable>

        <rollback>
            <dynamodb-pro:deleteDynamoTable tableName="deleteGlobalSecondaryIndexXml"/>
        </rollback>
    </changeSet>

    <changeSet id="2" author="your.name" labels="test">
        <dynamodb-pro:deleteGlobalSecondaryIndex tableName="deleteGlobalSecondaryIndexXml" indexName="globalIndex"/>

        <rollback/>
    </changeSet>

</databaseChangeLog>
